-- =================================================== -- OSM – SETUP SUPABASE (à exécuter une seule fois) -- =================================================== -- 1. Table profils clients create table if not exists public.profiles ( id uuid references auth.users on delete cascade primary key, email text, prenom text, nom text, societe text, metier text, created_at timestamptz default now() ); -- 2. Table favoris create table if not exists public.favorites ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users on delete cascade not null, track_id text not null, track_title text, track_playlist text, track_url text, created_at timestamptz default now(), unique(user_id, track_id) ); -- 3. Table téléchargements create table if not exists public.downloads ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users on delete cascade not null, track_id text not null, track_title text, track_playlist text, track_url text, downloaded_at timestamptz default now() ); -- 4. Table projets create table if not exists public.projects ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users on delete cascade not null, name text not null, description text, created_at timestamptz default now() ); -- 5. Activer RLS alter table public.profiles enable row level security; alter table public.favorites enable row level security; alter table public.downloads enable row level security; alter table public.projects enable row level security; -- 6. Politiques RLS (utilisateur voit ses propres données) create policy "profiles_own" on public.profiles for all using (auth.uid() = id); create policy "favorites_own" on public.favorites for all using (auth.uid() = user_id); create policy "downloads_own" on public.downloads for all using (auth.uid() = user_id); create policy "projects_own" on public.projects for all using (auth.uid() = user_id); -- 7. Trigger auto-création profil à l'inscription create or replace function public.handle_new_user() returns trigger language plpgsql security definer set search_path = public as $$ begin insert into public.profiles (id, email, prenom, nom, societe, metier) values ( new.id, new.email, new.raw_user_meta_data->>'prenom', new.raw_user_meta_data->>'nom', new.raw_user_meta_data->>'societe', new.raw_user_meta_data->>'metier' ) on conflict (id) do nothing; return new; end; $$; create or replace trigger on_auth_user_created after insert on auth.users for each row execute function public.handle_new_user(); -- 8. Fonction admin (contourne RLS avec mot de passe) create or replace function public.get_admin_data(admin_pwd text) returns json language plpgsql security definer set search_path = public, auth as $$ begin if admin_pwd != 'OSM_ADMIN_2026' then raise exception 'Accès refusé'; end if; return ( select json_build_object( 'profiles', ( select coalesce(json_agg( json_build_object( 'id', p.id, 'email', p.email, 'prenom', p.prenom, 'nom', p.nom, 'societe', p.societe, 'metier', p.metier, 'created_at', p.created_at, 'last_sign_in', u.last_sign_in_at ) order by p.created_at desc ), '[]'::json) from profiles p left join auth.users u on u.id = p.id ), 'favorites', ( select coalesce(json_agg(row_to_json(f.*) order by f.created_at desc), '[]'::json) from favorites f ), 'downloads', ( select coalesce(json_agg(row_to_json(d.*) order by d.downloaded_at desc), '[]'::json) from downloads d ) ) ); end; $$; grant execute on function public.get_admin_data(text) to anon;
